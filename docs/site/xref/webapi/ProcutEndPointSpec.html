24  @Path("/products")

27      @GET
28      @Path("/pricecalc")
29      @Produces(MediaType.APPLICATION_JSON)
30      @Operation(
31          operationId = "calculatePrice",
32          summary = "会員価格を計算する",
33          description = "会員の会員種別に応じた割引率を適用した価格を計算する")
34      @Parameters({
35          @Parameter(name = "memberNo", description = "会員番号。数値＋大文字アルファベットのみ", required = true,
36              schema = @Schema(implementation = String.class, minLength = 5, maxLength = 5)),
37          @Parameter(name = "price",  description = "価格", required = true,
38              schema = @Schema(implementation = Integer.class, minimum = "1", maximum = "99999999"))
39      })
40      @APIResponse(responseCode = "200",
41          description = "計算結果。1個目は割引後の価格、2個目は会員種別、3個目は割引率を格納",
42          content = @Content(mediaType = "application/json", schema =
43              @Schema(type = SchemaType.ARRAY, implementation = Integer.class, minItems = 3, maxItems = 3)))
44      List<Integer> calculatePrice(@QueryParam("memberNo") String memberNo, @QueryParam("price") int price);
45  }
